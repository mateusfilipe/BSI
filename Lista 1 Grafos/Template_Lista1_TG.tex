\begin{large}
	\textbf{Definições, Teoremas, Corolários, Observações, Lemas}
\end{large}

\vspace{0.5cm}

\textbf{Variáveis}: $G$ é o grafo; $N$ é o conjunto de vértices do grafo; $M$ é o conjunto de arestas do grafo; $d(x_i)$ é o grau do vértice $x_i$. 

\vspace{0.5cm}

%----------------------------------------------------------

\textcolor{red}{Tópico: Conceitos}

\begin{obs}
	Se todos os vértices de uma cadeia (ou passeio) forem distintos, a cadeia (ou passeio) será denominada \textbf{caminho}.
\end{obs}

\begin{obs}
	Quando o grafo $G$ é \textbf{orientado}, a sequência de arcos distintos que repete somente o primeiro e o último nó visitado se chama \textbf{circuito}.
\end{obs}

\begin{defn}
	A relação entre o número de arestas e vértices, muitas vezes denominado \textbf{densidade} de $G$, pode ser medida pela razão $\epsilon(G)$, tal que:
	$$\epsilon(G)=\dfrac{|M|}{|N|}=\dfrac{1}{2}d_M(G),$$
	onde $d_M(G)$ é denominada \textbf{média dos graus de $G$} e pode ser obtida por:
	$$d_M(G)=\dfrac{1}{|N|}=\sum\limits_{x_i \in N}d(x_i)$$
\end{defn}

\begin{lema}
	O somatório dos graus dos vértices de um grafo é igual a $2m$, onde $m$ representa o número de arestas do grafo.
\end{lema}

\begin{defn}
	Um grafo $G$ é dito par se \textbf{todos os seus vértices possuírem grau par}. 
\end{defn}

\begin{obs}
	Um grafo com todos os vértices possuindo grau ímpar \textbf{não é um grafo ímpar}.
\end{obs}

\begin{obs}
	\textbf{Maximal} deve ser distinto de \textbf{máximo}. \textbf{ Maximal} é condição de pertinência.
\end{obs}

\begin{obs}
	\textbf{Grafos conexos} possuem apenas uma \textbf{componente conexa}.
\end{obs}

\begin{defn}
	O \emph{rank} ``r'', ou \textbf{posto}, de um grafo  $G$ com $n$ vértices e $c$ componentes conexas é dado por:
	$$r=n-c$$
\end{defn}

\begin{defn}
	\textbf{Nulidade} $L$ de um grafo $G$ com $m$ arestas, $n$ vértices e $c$ componentes conexas, é definida como:
	$$L=m-n+c=m-r$$
\end{defn}

\begin{teo}
	Um grafo $G$ bipartido se e somente se todo ciclo em $G$ for par.
\end{teo}

\begin{teo}
	O número de arestas em um grafo completo $G=(N,M)$ é
	$$\dfrac{n(n-1)}{2},$$
	onde $n=|N|$
\end{teo}

\begin{obs}
	Um grafo regular com grau \textbf{três} é denominado \textbf{cúbico}. Um grafo com grau \textbf{quatro} é denominado \textbf{quartic}.
\end{obs}

\begin{teo}
	O número de vértices de grau ímpar de um grafo é sempre par.
\end{teo}

\begin{teo}[Teorema de Euler]
	$$n+f-m=2,$$
	onde $n=|N|,~m=|M|$ e $f$ denota o número de regiões.
\end{teo}

\begin{cor}[Corolário do Teorema de Euler]
	$$m \le 3n-6,$$
	onde $n=|N|$ e $m=|M|$.
\end{cor}

\begin{defn}
	Um grafo $H$ é chamado \textbf{minor} - ou \textbf{menor} - de um grafo $G$ se $H$ é isomorfo a um grafo que pode ser obtido por uma sequência finita de \textbf{contrações de arestas} de $G$.
\end{defn}

\begin{teo}
	Um grafo $G$ é planar se e somente se não possui $K_5$ ou ao $K_{3,3}$ como minor.
\end{teo}

\begin{obs}
	A adição de arestas é também denominada de \textbf{join}.
\end{obs}

\begin{obs}
	A \textbf{matriz de adjacência} ocupa $O(n^2)$ posições de memória.
\end{obs}

\begin{obs}
	A \textbf{matriz de incidência} ocupa $O(nm)$ posições de memória.
\end{obs}

\begin{obs}
	A \textbf{lista de adjacência} ocupa $O(n+m)$ posições de memória.
\end{obs}

\begin{obs}
	A \textbf{representação vetorial} ocupa $O(m+n)$ posições de memória.
\end{obs}

\textcolor{red}{Tópico: Árvores}

\begin{teo}
	Em um grafo $G$ o número de subgrafos árvores com $n$ vértices e com grau especificado $k$, é igual a:
	$$n(T^k)={n-2\choose k-1}(n-1)^{n-k-1}$$
\end{teo}

\begin{teo}[Fórmula de Caley]
	O número de árvores distintas em um grafo completo com $n$ vértices é $n^{n-2}$.
\end{teo}

\begin{obs}
	Todo grafo conexo possui \textbf{pelo menos uma árvore geradora}.
\end{obs}

\begin{defn}
	Uma floresta é \textbf{um conjunto de árvores} sem vértices em comum. \\
	Uma floresta geradora é uma floresta que contém \textbf{todos os vértices} de $G$.
\end{defn}

\begin{obs}
	A determinação das árvores geradoras mínima ou máxima pode ser feita em tempo polinomial.
\end{obs}

\begin{obs}
	O número de vértices em uma \textbf{árvore binária cheia é $2^{h+1}-1$}, onde $h$ é altura da árvore.
\end{obs}

\item[] Um caminho segundo a Teoria dos Grafos é:
\begin{itemize}
	\item[(~\textbf{V}~)] qualquer conexão entre dois vértices. 
	\item[(~\textbf{V}~)] uma sequência de arestas contínua sem vértices repetidos. 
	\item[(~\textbf{V}~)] uma sequência de arestas sem repetição de arestas ou vértices. 
	\item[(~\textbf{F}~)] qualquer sequência de arestas, desde que seja aberta. 
\end{itemize}

\item[] Um grafo é conexo se:
\begin{itemize}
	\item[(~\textbf{V}~)] para qualquer vértice pertencente ao grafo sempre for possível encontrar um caminho até outro vértice distinto e também pertencente ao grafo. 
	\item[(~\textbf{F}~)] para qualquer vértice do grafo sempre é possível encontrar um ciclo. 
	\item[(~\textbf{F}~)] o grafo for simples e o número de arestas for igual a pelo menos $n+1$. 
	\item[(~\textbf{F}~)] o menor grau dos vértices do grafo for dois. 
\end{itemize}

\item[] Um grafo será planar se:
\begin{itemize}
	\item[(~\textbf{V}~)] não contiver como subgrafo um $K_5$ ou um $K_{3,3}$. 
	\item[(~\textbf{F}~)] o grau máximo de seus vértices for 3. 
	\item[(~\textbf{F}~)] for isomorfo ao grafo de Petersen. 
	\item[(~\textbf{F}~)] for um subgrafo do grafo de Petersen. 
\end{itemize}

\item[] De forma geral
\begin{itemize}
	\item[(~\textbf{V}~)] o número de vértices de um grafo com todos os vértices com grau ímpar é sempre par. 
	\item[(~\textbf{F}~)] é impossível existir um grafo com um número par de vértices com grau par se nesse grafo existirem dois vértices com grau ímpar. 
	\item[(~\textbf{F}~)] um grafo desconexo deve possuir pelo menos duas cinturas. 
	\item[(~\textbf{F}~)] um grafo desconexo não possui cintura. 
\end{itemize}
